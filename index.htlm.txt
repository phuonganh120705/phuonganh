    }

<body>
  <div id="map"></div>
  <script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>
  <script>
    var map = L.map('map').setView([21.0285, 105.8542], 11);

    // Nền OpenStreetMap
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '&copy; OpenStreetMap contributors'
    }).addTo(map);

    // Hàm màu theo Natural Breaks dữ liệu bạn cung cấp
    function getColor(d) {
      return d > 10000 ? '#800026' :
             d > 8000  ? '#BD0026' :
             d > 5000  ? '#E31A1C' :
             d > 3000  ? '#FC4E2A' :
             d > 1000  ? '#FD8D3C' :
                         '#FEB24C';
    }

    function style(feature) {
      return {
        fillColor: getColor(feature.properties.Enterprises_Under_10),
        weight: 2,
        opacity: 1,
        color: 'white',
        dashArray: '3',
        fillOpacity: 0.7
      };
    }

    function onEachFeature(feature, layer) {
      if (feature.properties) {
        layer.bindPopup(
          '<b>' + feature.properties.District_Name + '</b><br>' +
          'Enterprises <10 employees: ' + feature.properties.Enterprises_Under_10
        );
      }
    }

    // Load GeoJSON dữ liệu
    fetch('Hanoi_Districts_Under10.geojson')
      .then(res => res.json())
      .then(data => {
        L.geoJson(data, {
          style: style,
          onEachFeature: onEachFeature
        }).addTo(map);
      });

    // Legend
    var legend = L.control({position: 'topright'});

    legend.onAdd = function(map) {
      var div = L.DomUtil.create('div', 'legend'),
          grades = [0,1000,3000,5000,8000,10000],
          labels = [];

      div.innerHTML = '<b>Number of enterprises</b><br>';
      for (var i = 0; i < grades.length; i++) {
          div.innerHTML +=
              '<i style="background:' + getColor(grades[i] + 1) + '"></i> ' +
              grades[i] + (grades[i + 1] ? '&ndash;' + grades[i + 1] + '<br>' : '+');
      }
      return div;
    };

    legend.addTo(map);
  </script>
